require_relative 'openrouter_client'

# Inicializar o cliente
client = Ai::OpenRouter::Assistant.new(
  api_key: 'sua-chave-api-aqui',
  site_url: 'https://seu-site.com',  # Opcional
  site_name: 'Seu App'                # Opcional
)

# ============================================
# EXEMPLO 1: Requisição simples (não-streaming)
# ============================================
puts "=== EXEMPLO 1: Requisição Simples ==="

response = client.chat_completion(
  model: 'openai/gpt-4o',
  messages: [
    { role: 'user', content: 'Qual é o sentido da vida?' }
  ]
)

puts "Resposta: #{response[:choices][0][:message][:content]}"
puts "Tokens usados: #{response[:usage][:total_tokens]}"
puts "ID da geração: #{response[:id]}"
puts "\n"

# ============================================
# EXEMPLO 2: Com parâmetros avançados
# ============================================
puts "=== EXEMPLO 2: Com Parâmetros Avançados ==="

response = client.chat_completion(
  model: 'anthropic/claude-3.5-sonnet',
  messages: [
    { role: 'system', content: 'Você é um assistente prestativo e conciso.' },
    { role: 'user', content: 'Explique inteligência artificial em 3 frases.' }
  ],
  temperature: 0.7,
  max_tokens: 150,
  top_p: 0.9
)

puts "Resposta: #{response[:choices][0][:message][:content]}"
puts "\n"

# ============================================
# EXEMPLO 3: Streaming
# ============================================
puts "=== EXEMPLO 3: Streaming ==="

client.chat_completion(
  model: 'openai/gpt-4o',
  messages: [
    { role: 'user', content: 'Conte uma história curta sobre um robô.' }
  ],
  stream: true
) do |chunk|
  content = chunk.dig(:choices, 0, :delta, :content)
  print content if content
end

puts "\n\n"

# ============================================
# EXEMPLO 4: Usando helper para mensagens com imagem

puts "=== EXEMPLO 4: Mensagem com Imagem ==="

image_message = Ai::OpenRouter::Assistant.create_image_message(

# ============================================
# EXEMPLO 5: Usando ferramentas (tools)
# ============================================
puts "=== EXEMPLO 5: Tool Calling ==="

# Criar uma ferramenta
weather_tool = Ai::OpenRouter::Assistant.create_tool(
  name: 'get_weather',
  description: 'Obtém a previsão do tempo para uma localização',
  parameters: {
    type: 'object',
    properties: {
      location: {
        type: 'string',
        description: 'A cidade e estado, ex: São Paulo, SP'
      },
      unit: {
        type: 'string',
        enum: ['celsius', 'fahrenheit'],
        description: 'A unidade de temperatura'
      }
    },
    required: ['location']
  }
)

response = client.chat_completion(
  model: 'openai/gpt-4o',
  messages: [
    { role: 'user', content: 'Qual é a temperatura em São Paulo?' }
  ],
  tools: [weather_tool],
  tool_choice: 'auto'
)

if response[:choices][0][:message][:tool_calls]
  tool_call = response[:choices][0][:message][:tool_calls][0]
  puts "Ferramenta chamada: #{tool_call[:function][:name]}"
  puts "Argumentos: #{tool_call[:function][:arguments]}"
end
puts "\n"
